def calcular_tensao(forca, area):
    """
    Calcula a tensão mecânica.

    Parâmetros:
    forca (float): força aplicada em Newton
    area (float): área em metros quadrados

    Retorna:
    float: tensão em Pascal
    """
    return forca / area
